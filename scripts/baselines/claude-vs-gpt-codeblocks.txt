```python
import anthropic

client = anthropic.Anthropic()
message = client.messages.create(
    model="claude-sonnet-4-6-20250514",
    max_tokens=1024,
    system="당신은 시니어 백엔드 엔지니어입니다.",
    messages=[
        {"role": "user", "content": "REST API 설계 원칙을 설명해주세요."}
    ]
)
```
